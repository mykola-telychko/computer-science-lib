Є масив виміряних значень. Спочатку знайти середнє значення, 
потім за допомогою map порахувати квадрати відхилень і звести їх через reduce.
const arr = [10, 12, 14, 16, 18];
// 1. Calculate the mean (average) of the data
const mean = arr.reduce((finSum, num) => finSum + num, 0) / arr.length;
// 2. Calculate squared deviations from the mean for each data point
const sqrDeviations = arr.map(num => Math.pow(num - mean, 2));
// 3. Sum all squared deviations
const sumSqrDeviations = sqrDeviations.reduce((finSum, deviation) => finSum + deviation, 0);
// Output results
console.log("Mean (Average):", mean); // 14
console.log("Squared Deviations:", sqrDeviations); // [16, 4, 0, 4, 16]
console.log("Sum of Squared Deviations:", sumSqrDeviations); // 40

Обчислення інтеграла (метод прямокутників)
Є функція f(x). Створити дискретний набір точок (наприклад, від 0 до 10 з кроком 0.1). 
За допомогою map обчислити f(x) і через reduce знайти площу під графіком.
--------
// Функція, інтеграл якої обчислюємо (наприклад, f(x) = x^2)
const f = (x) => x ** 2;

// Межі інтегрування та крок
const a = 0; // Початок відрізку
const b = 10; // Кінець відрізку
const h = 0.1; // Крок

// Створюємо масив точок x від a до b з кроком h
const points = [];
for (let x = a; x <= b; x += h) { points.push(x); }

// значення функції f(x) у точках
const functionValues = points.map(x => f(x));

// Метод лівих прямокутників: сумуємо f(x_i) * h, крім останньої точки
const integralLeft = functionValues
    .slice(0, -1) // Виключаємо останню точку, щоб не виходити за межі
    .reduce((sum, current) => sum + current * h, 0);

// Метод правих прямокутників: сумуємо f(x_{i+1}) * h, починаючи з другої точки
const integralRight = functionValues
    .slice(1) // Виключаємо першу точку
    .reduce((sum, current) => sum + current * h, 0);

// Метод середніх прямокутників (точніший):
// Зсуваємо точки на h/2 вправо і обчислюємо значення функції
const shiftedPoints = points.slice(0, -1).map(x => x + h / 2);
const functionValuesMid = shiftedPoints.map(x => f(x));
const integralMid = functionValuesMid.reduce((sum, current) => sum + current * h, 0);

console.log("Інтеграл (ліві прямокутники):", integralLeft);
console.log("Інтеграл (праві прямокутники):", integralRight);
console.log("Інтеграл (метод середніх прямокутників):", integralMid);

Середня кінетична енергія
Є масив швидкостей частинок. Використати map для обчислення кінетичної 
енергії E = (m * v²)/2 для кожної частинки (m однакова), а потім reduce для знаходження середнього значення.
-----------
// Масив швидкостей частинок (м/с)
const velocities = [10, 15, 20, 8, 12, 25, 18, 22, 14, 16];

// Маса частинки (кг) - однакова для всіх частинок
const mass = 0.5;

// Функція для обчислення кінетичної енергії: E = (m * v²) / 2
function calculateKineticEnergy(velocity) {
    return (mass * velocity * velocity) / 2;
}

console.log('=== Обчислення кінетичної енергії частинок ===');
console.log(`Масив швидкостей: [${velocities.join(', ')}] м/с`);
console.log(`Маса кожної частинки: ${mass} кг`);
console.log();
// Використання map для обчислення кінетичної енергії кожної частинки
const kineticEnergies = velocities.map(calculateKineticEnergy);

console.log('Кінетичні енергії кожної частинки:');
kineticEnergies.forEach((energy, index) => {
    console.log(`Частинка ${index + 1}: v = ${velocities[index]} м/с → E = ${energy.toFixed(2)} Дж`);
});
console.log();
console.log(`Масив кінетичних енергій: [${kineticEnergies.map(e => e.toFixed(2)).join(', ')}] Дж`);

// Використання reduce для знаходження середнього значення
const averageKineticEnergy = kineticEnergies.reduce((sum, energy, index, array) => {
    return sum + energy / array.length;
}, 0);

// Альтернативний спосіб через reduce для суми та окреме ділення
const totalEnergy = kineticEnergies.reduce((sum, energy) => sum + energy, 0);
const averageEnergy2 = totalEnergy / kineticEnergies.length;

console.log();
console.log('=== Результати ===');
console.log(`Загальна кінетична енергія: ${totalEnergy.toFixed(2)} Дж`);
console.log(`Кількість частинок: ${kineticEnergies.length}`);
console.log(`Середня кінетична енергія: ${averageKineticEnergy.toFixed(2)} Дж`);

// Перевірка результату
console.log(`Перевірка (альтернативний розрахунок): ${averageEnergy2.toFixed(2)} Дж`);

console.log();

// Додаткова статистика
const minEnergy = Math.min(...kineticEnergies);
const maxEnergy = Math.max(...kineticEnergies);
console.log('=== Додаткова статистика ===');
console.log(`Мінімальна кінетична енергія: ${minEnergy.toFixed(2)} Дж`);
console.log(`Максимальна кінетична енергія: ${maxEnergy.toFixed(2)} Дж`);

Момент сили (Torque)
Є масив сил та відповідних плечей (відстаней до точки обертання). 
Використати map для обчислення моментів M = F * r і reduce для суми всіх моментів.
----------
// Масиви сил і плечей
const forces = [10, 15, 7, 20];   // в Ньютонах
const arms   = [0.5, 0.3, 0.8, 0.6]; // відстані (м)

// Обчислюємо моменти: M = F * r
const torques = forces.map((F, i) => F * arms[i]);
console.log("Окремі моменти:", torques);
// Знаходимо суму моментів totalTorque
console.log("Сумарний момент сили:", torques.reduce((sum, M) => sum + M, 0), "Н·м");

Сумарний електричний заряд системи
Є набір частинок із зарядами. За допомогою reduce знайти сумарний заряд,
 а через map можна додатково вирахувати співвідношення заряду до маси для кожної частинки.
// Array of particles with charge (in Coulombs) and mass (in kilograms)
const particles = [
  { charge: 1.6e-19, mass: 9.1e-31 },  // Electron
  { charge: -1.6e-19, mass: 9.1e-31 }, // Another electron
  { charge: 3.2e-19, mass: 6.6e-27 },  // Alpha particle
];
// 1. Calculate total electric charge using reduce // chargeToMassRatios = particles.map
const totalCharge = particles.reduce((sum, particle) => sum + particle.charge, 0);
// 2. Calculate charge-to-mass ratio for each particle using map
const chargeToMassRatios = particles.map(particle => ({
  charge: particle.charge,
  mass: particle.mass,
  ratio: particle.charge / particle.mass
}));
console.log("Total Electric Charge (C):", totalCharge);
console.log("Charge-to-Mass Ratios (C/kg):", chargeToMassRatios);


Середнє значення сили по часу
Є масив вимірювань сили у різні моменти часу. Використати map для перетворення 
у імпульси F * Δt, а потім reduce для знаходження середньої сили.
// Array of force measurements (in Newtons) at different time intervals
const forceMeasurements = [
  { force: 10, timeInterval: 2 },  // Force of 10N applied for 2 seconds
  { force: 15, timeInterval: 1 },  // Force of 15N applied for 1 second
  { force: 8, timeInterval: 3 },   // Force of 8N applied for 3 seconds
  { force: 12, timeInterval: 2 }   // Force of 12N applied for 2 seconds
];
// 1. Calculate impulse (F * Δt) for each measurement using map
const impulses = forceMeasurements.map(measurement => ({
  force: measurement.force,
  timeInterval: measurement.timeInterval,
  impulse: measurement.force * measurement.timeInterval
}));
// 2. Calculate total impulse and total time using reduce
const total = forceMeasurements.reduce((acc, measurement) => ({
  totalImpulse: acc.totalImpulse + (measurement.force * measurement.timeInterval),
  totalTime: acc.totalTime + measurement.timeInterval
}), { totalImpulse: 0, totalTime: 0 });
console.log("Individual Impulses (N·s):", impulses);
console.log("Total Impulse:", total.totalImpulse, "N·s");
console.log("Total Time:", total.totalTime, "s");
// 3. Calculate average force (total impulse / total time)
console.log("Average Force:", total.totalImpulse / total.totalTime, "N");



Потужність системи
Є масив швидкостей та відповідних сил. За допомогою map порахувати миттєву 
потужність P = F * v, а потім reduce для знаходження середньої потужності.
// Array of velocity (m/s) and force (N) measurements
const measurements = [
  { velocity: 5, force: 10 },   // v = 5 m/s, F = 10 N
  { velocity: 8, force: 12 },   // v = 8 m/s, F = 12 N
  { velocity: 6, force: 15 },   // v = 6 m/s, F = 15 N
  { velocity: 7, force: 18 }    // v = 7 m/s, F = 18 N
];
// 1. Calculate instantaneous power (P = F * v) for each measurement using map
const instantaneousPowers = measurements.map(measurement => ({
  velocity: measurement.velocity,
  force: measurement.force,
  power: measurement.force * measurement.velocity  // P = F * v
}));
// 2. Calculate total power and count measurements using reduce
const powerSummary = measurements.reduce((accumulator, measurement) => ({
  totalPower: accumulator.totalPower + (measurement.force * measurement.velocity),
  measurementCount: accumulator.measurementCount + 1
}), { totalPower: 0, measurementCount: 0 });
console.log("Instantaneous Powers (W):", instantaneousPowers);
console.log("Total Power:", powerSummary.totalPower, "W");
console.log("Number of Measurements:", powerSummary.measurementCount);
// 3. Calculate average power
console.log("Average Power:", powerSummary.totalPower / powerSummary.measurementCount, "W");
